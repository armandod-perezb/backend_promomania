Asegurate que la estructura del backend esté constituida de esta manera, actualiza los models y las vistas correspondientes:

-- MODELO RELACIONAL (alineado a tu app + Django API)
-- PostgreSQL-style SQL (ajustable a MySQL)

-- =========================
-- 1) USUARIOS
-- =========================
CREATE TABLE usuarios (
  id                BIGSERIAL PRIMARY KEY,
  nombre            VARCHAR(120) NOT NULL,
  correo            VARCHAR(150) NOT NULL UNIQUE,
  password          VARCHAR(255) NOT NULL,
  ciudad            VARCHAR(120),
  nivel             INTEGER,
  puntuacion        INTEGER,
  rol               VARCHAR(20) NOT NULL DEFAULT 'usuario',   -- admin | usuario
  estado            VARCHAR(20) NOT NULL DEFAULT 'activo',    -- activo | inactivo
  created_at        TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at        TIMESTAMP NOT NULL DEFAULT NOW()
);

-- =========================
-- 2) CATÁLOGOS
-- =========================
CREATE TABLE supermercados (
  id                BIGSERIAL PRIMARY KEY,
  nombre            VARCHAR(120) NOT NULL,
  direccion         VARCHAR(200),
  ciudad            VARCHAR(120),
  estado            VARCHAR(20) NOT NULL DEFAULT 'activo',    -- activo | inactivo
  created_at        TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at        TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE TABLE categorias (
  id                BIGSERIAL PRIMARY KEY,
  nombre            VARCHAR(120) NOT NULL UNIQUE,
  descripcion       TEXT
);

CREATE TABLE tipos_promocion (
  id                BIGSERIAL PRIMARY KEY,
  nombre            VARCHAR(120) NOT NULL UNIQUE,
  descripcion       TEXT,
  estado            VARCHAR(20) NOT NULL DEFAULT 'activo'     -- activo | inactivo
);

-- =========================
-- 3) PROMOCIONES
-- =========================
CREATE TABLE promociones (
  codigo            VARCHAR(40) PRIMARY KEY,
  titulo            VARCHAR(180) NOT NULL,
  descripcion       TEXT,
  precio            NUMERIC(12,2) NOT NULL CHECK (precio >= 0),
  descuento         INTEGER CHECK (descuento IS NULL OR (descuento >= 0 AND descuento <= 100)),
  condicion_producto VARCHAR(30) NOT NULL DEFAULT 'nuevo',    -- nuevo | usado | reacondicionado
  ubicacion         VARCHAR(200),
  url               TEXT,
  foto              TEXT,
  foto_es_local     BOOLEAN NOT NULL DEFAULT FALSE,
  tipo_vigencia     VARCHAR(20) NOT NULL DEFAULT 'por_fecha', -- por_fecha | permanente
  fecha_inicio      DATE,
  fecha_fin         DATE,
  estado            VARCHAR(20) NOT NULL DEFAULT 'pendiente', -- pendiente | aprobada | rechazada
  vistas            INTEGER NOT NULL DEFAULT 0 CHECK (vistas >= 0),
  lat               NUMERIC(10,7),
  lng               NUMERIC(10,7),

  id_usuario        BIGINT NOT NULL,
  id_supermercado   BIGINT NOT NULL,
  id_categoria      BIGINT NOT NULL,
  id_tipo_promocion BIGINT NOT NULL,

  created_at        TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at        TIMESTAMP NOT NULL DEFAULT NOW(),

  CONSTRAINT fk_promocion_usuario
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id),
  CONSTRAINT fk_promocion_supermercado
    FOREIGN KEY (id_supermercado) REFERENCES supermercados(id),
  CONSTRAINT fk_promocion_categoria
    FOREIGN KEY (id_categoria) REFERENCES categorias(id),
  CONSTRAINT fk_promocion_tipo
    FOREIGN KEY (id_tipo_promocion) REFERENCES tipos_promocion(id),
  CONSTRAINT chk_fechas_promocion
    CHECK (fecha_fin IS NULL OR fecha_inicio IS NULL OR fecha_fin >= fecha_inicio)
);

CREATE INDEX idx_promociones_estado ON promociones(estado);
CREATE INDEX idx_promociones_categoria ON promociones(id_categoria);
CREATE INDEX idx_promociones_supermercado ON promociones(id_supermercado);
CREATE INDEX idx_promociones_usuario ON promociones(id_usuario);

-- =========================
-- 4) HORARIOS DE PROMOCIÓN
-- =========================
CREATE TABLE promociones_horarios (
  id                BIGSERIAL PRIMARY KEY,
  dia_semana        VARCHAR(15) NOT NULL,                     -- lunes..domingo
  hora_inicio       TIME NOT NULL,
  hora_fin          TIME NOT NULL,
  codigo_promocion  VARCHAR(40) NOT NULL,

  CONSTRAINT fk_horario_promocion
    FOREIGN KEY (codigo_promocion) REFERENCES promociones(codigo) ON DELETE CASCADE,
  CONSTRAINT chk_horario_rango
    CHECK (hora_fin > hora_inicio)
);

CREATE INDEX idx_horarios_promocion ON promociones_horarios(codigo_promocion);

-- =========================
-- 5) COMENTARIOS
-- =========================
CREATE TABLE comentarios (
  id                BIGSERIAL PRIMARY KEY,
  contenido         TEXT NOT NULL,
  fecha             TIMESTAMP NOT NULL DEFAULT NOW(),
  id_usuario        BIGINT NOT NULL,
  codigo_promocion  VARCHAR(40) NOT NULL,
  id_comment_reply  BIGINT NULL,

  CONSTRAINT fk_comentario_usuario
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id),
  CONSTRAINT fk_comentario_promocion
    FOREIGN KEY (codigo_promocion) REFERENCES promociones(codigo) ON DELETE CASCADE,
  CONSTRAINT fk_comentario_reply
    FOREIGN KEY (id_comment_reply) REFERENCES comentarios(id) ON DELETE CASCADE
);

CREATE INDEX idx_comentarios_promocion ON comentarios(codigo_promocion);
CREATE INDEX idx_comentarios_usuario ON comentarios(id_usuario);

-- =========================
-- 6) VALORACIONES (likes/dislikes)
-- =========================
CREATE TABLE valoraciones (
  id                BIGSERIAL PRIMARY KEY,
  tipo              VARCHAR(20) NOT NULL,                     -- positiva | negativa
  id_usuario        BIGINT NOT NULL,
  codigo_promocion  VARCHAR(40) NOT NULL,

  CONSTRAINT fk_valoracion_usuario
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id),
  CONSTRAINT fk_valoracion_promocion
    FOREIGN KEY (codigo_promocion) REFERENCES promociones(codigo) ON DELETE CASCADE,
  CONSTRAINT uq_valoracion_usuario_promocion
    UNIQUE (id_usuario, codigo_promocion),
  CONSTRAINT chk_tipo_valoracion
    CHECK (tipo IN ('positiva', 'negativa'))
);

CREATE INDEX idx_valoraciones_promocion ON valoraciones(codigo_promocion);

-- =========================
-- 7) FAVORITOS
-- =========================
CREATE TABLE favoritos (
  id                BIGSERIAL PRIMARY KEY,
  fecha             TIMESTAMP NOT NULL DEFAULT NOW(),
  id_usuario        BIGINT NOT NULL,
  codigo_promocion  VARCHAR(40) NOT NULL,

  CONSTRAINT fk_favorito_usuario
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id),
  CONSTRAINT fk_favorito_promocion
    FOREIGN KEY (codigo_promocion) REFERENCES promociones(codigo) ON DELETE CASCADE,
  CONSTRAINT uq_favorito_usuario_promocion
    UNIQUE (id_usuario, codigo_promocion)
);

CREATE INDEX idx_favoritos_usuario ON favoritos(id_usuario);
CREATE INDEX idx_favoritos_promocion ON favoritos(codigo_promocion);

-- =========================
-- 8) REPORTES
-- =========================
CREATE TABLE reportes (
  id                BIGSERIAL PRIMARY KEY,
  motivo            TEXT NOT NULL,
  fecha             TIMESTAMP NOT NULL DEFAULT NOW(),
  estado            VARCHAR(20) NOT NULL DEFAULT 'pendiente', -- pendiente | revisado | descartado
  id_usuario        BIGINT NOT NULL,
  codigo_promocion  VARCHAR(40) NOT NULL,

  CONSTRAINT fk_reporte_usuario
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id),
  CONSTRAINT fk_reporte_promocion
    FOREIGN KEY (codigo_promocion) REFERENCES promociones(codigo) ON DELETE CASCADE
);

CREATE INDEX idx_reportes_estado ON reportes(estado);
CREATE INDEX idx_reportes_promocion ON reportes(codigo_promocion);

-- =========================
-- 9) NOTIFICACIONES (opcional, si la vas a persistir)
-- =========================
CREATE TABLE notificaciones (
  id                BIGSERIAL PRIMARY KEY,
  mensaje           TEXT NOT NULL,
  fecha             TIMESTAMP NOT NULL DEFAULT NOW(),
  tipo              VARCHAR(40) NOT NULL,
  id_usuario_destino BIGINT NULL,
  leida             BOOLEAN NOT NULL DEFAULT FALSE,

  CONSTRAINT fk_notificacion_usuario
    FOREIGN KEY (id_usuario_destino) REFERENCES usuarios(id) ON DELETE SET NULL
);

CREATE INDEX idx_notificaciones_usuario ON notificaciones(id_usuario_destino);
CREATE INDEX idx_notificaciones_leida ON notificaciones(leida);
